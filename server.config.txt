server {
  charset utf-8;
    server_name graffitee.ml www.graffitee.ml; # managed by Certbot


  # angular app & front-end files
  location / {
    proxy_pass http://localhost:8000/;
    proxy_set_header   X-Forwarded-For $remote_addr;
    proxy_set_header   Host $http_host;
    proxy_http_version 1.1;
    proxy_set_header   Upgrade $http_upgrade;
    proxy_set_header   Connection "upgrade";
  }

  # node api reverse proxy


    listen 443 ssl; # managed by Certbot
    ssl_certificate /etc/letsencrypt/live/graffitee.ml/fullchain.pem; # managed by Certbot
        ssl_certificate_key /etc/letsencrypt/live/graffitee.ml/privkey.pem; # managed by Certbot
    include /etc/letsencrypt/options-ssl-nginx.conf; # managed by Certbot
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem; # managed by Certbot



}

server {
    if ($host = www.graffitee.ml) {
        return 301 https://$host$request_uri;
    } # managed by Certbot


    if ($host = graffitee.ml) {
        return 301 https://$host$request_uri;
    } # managed by Certbot


  listen 80 ;
    server_name graffitee.ml www.graffitee.ml;
    return 404; # managed by Certbot




}
